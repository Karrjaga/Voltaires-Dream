#establishment of county of trinacria
country_event = {
	id = vn_flavor_apl.102
	title = "vn_flavor_apl.102.t"
	desc = "vn_flavor_apl.102.d"
	picture = CONQUEST_eventPicture

	is_triggered_only = yes
	major = yes
	major_trigger = {
		capital_scope = {
			OR = {
				superregion = italy_superregion
				superregion = sicily_superregion
			}
		}
	}

	option = {
		name = vn_flavor_apl.102.a
		ai_chance = {
			factor = 1
		}
		hidden_effect = {
			sicilia_region = {
				limit = { owned_by = ROOT }
				add_core = TNC
			}
		}
		release = TNC
		sicilia_region = {
			remove_core = ROOT
			add_permanent_claim = TNC
		}
		if = {
			limit = { has_dlc = "Winds of Change" }
			create_subject = {
				subject_type = incorporated_vassal
				who = TNC
			}
		}
		else = {
			create_subject = {
				subject_type = vassal
				who = TNC
			}
		}
		hidden_effect = {
			TNC = {
				add_liberty_desire = -50
				change_government = monarchy
				add_government_reform = siculo_norman_monarchy
				add_treasury = 100
				add_manpower = 5
				define_ruler = {
					name = "Roger I"
					dynasty = "de Hauteville"
					culture = normand
					adm = 3
					dip = 4
					mil = 5
				}
				define_heir = {}
				define_ruler_to_general = {
					fire = 4
					shock = 4
					manuever = 3
					siege = 3
				}
				add_ruler_personality = strict_personality
				add_ruler_personality = kind_hearted_personality
				add_ruler_personality = tolerant_personality
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.8
						cavalry = 0.2
					}
				}
			}
		}
	}
	option = {
		name = vn_flavor_apl.102.b
		ai_chance = {
			factor = 0
		}
		add_country_modifier = {
			name = apl_foothold_in_sicily_mission
			duration = 9125
		}
	}
}
