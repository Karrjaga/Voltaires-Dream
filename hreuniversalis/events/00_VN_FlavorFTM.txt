	title = "vn_flavor_ftm.1.t"

	fire_only_once = yes
	is_triggered_only = yes
	major = yes
	major_trigger = {
		religion_group = muslim
	}

	trigger = {
		normal_or_historical_nations = yes
		current_age = age_of_empire
		tag = FTM
		government = monarchy
		religion = shiite
		#has_heir_flag = nizar_heir_flag
		NOT = { started_in = 1094.1.10 }
		num_of_cities = 10
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			set_country_flag = avoid_choosing_madhhab_flag
		}
	}

	option = {
		name = "vn_flavor_ftm.1.a"
		ai_chance = { factor = 90 }
		if = {
			limit = { has_government_mechanic = fatimid_viziers_vs_caliph_mechanic }
			add_government_power = {
				mechanic_type = fatimid_viziers_vs_caliph_mechanic
				power_type = caliphal_authority
				value = -5
			}
		}
		add_stability = -1
		define_ruler = { dynasty = ROOT	}
		if = {
			limit = { has_heir_flag = nizar_heir_flag }
			kill_heir = { allow_new_heir = no }
		}
		if = {
			limit = {
				5022 = {
					is_capital = no
					controlled_by = ROOT
				}
			}
			5022 = {
				spawn_rebels = {
					size = 3
					type = pretender_rebels
					win = yes
					religion = qarmatian
					leader = "al-Mustafâ"
					leader_dynasty = "Banû Fâtimah"
				}
			}
		}
		else = {
			random_owned_province = {
				limit = {
					is_capital = no
					controlled_by = ROOT
				}
				spawn_rebels = {
					size = 4
					type = pretender_rebels
					win = yes
					religion = qarmatian
					leader = "al-Mustafâ"
					leader_dynasty = "Banû Fâtimah"
				}
			}
		}
		clr_country_flag = avoid_choosing_madhhab_flag
	}
	option = {
		name = "vn_flavor_ftm.1.b"
		ai_chance = { factor = 10 }
		if = {
			limit = { has_government_mechanic = fatimid_viziers_vs_caliph_mechanic }
			add_government_power = {
				mechanic_type = fatimid_viziers_vs_caliph_mechanic
				power_type = caliphal_authority
				value = 5
			}
		}
		add_stability = 1
		change_religion = qarmatian
		set_religious_school = {
			group = muslim
			school = nizari_school
		}
		random_owned_province = {
			limit = {
				is_capital = no
				controlled_by = ROOT
				OR = {
					region = lower_egypt_region
					region = middle_egypt_region
				}
			}
			spawn_rebels = {
				size = 4
				type = pretender_rebels
				win = yes
				religion = shiite
				leader = "al-Musta‘lî"
				leader_dynasty = "Banû Fâtimah"
			}
		}
		clr_country_flag = avoid_choosing_madhhab_flag
	}
}
